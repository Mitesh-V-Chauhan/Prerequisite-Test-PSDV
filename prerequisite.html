# Sort the remaining words alphabetically.

    words= s.split()

    word_lengths = {word: len(word) for word in sorted_words}

def fnct(l1, l2):
    intersection=[]
    union=[]
    symmetric_difference=[]
    for el in l1:
        if el in l2:
            intersection.append(el)
    union= list(set(l1 + l2))
    symmetric_difference=list(set(l1 + l2))
    for elm in union:
        if elm in intersection:
            symmetric_difference.remove(elm)
    print(f'''{intersection}
{union}
{symmetric_difference}''')         
    return
